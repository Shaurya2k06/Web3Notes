# Smart Contracts

## What are Smart Contracts?

Smart contracts are self-executing programs stored on a blockchain that automatically enforce and execute agreements when predetermined conditions are met. They eliminate the need for intermediaries and enable trustless transactions.

### Definition

**Smart Contract**: A computer program that:
1. Runs on a blockchain
2. Contains business logic and state
3. Executes automatically when conditions are met
4. Is immutable once deployed
5. Is transparent and verifiable

## Key Characteristics

### 1. Distributed
Replicated across all network nodes.

### 2. Deterministic
Same input always produces same output across all nodes.

### 3. Immutable
Once deployed, code cannot be changed (unless upgrade mechanism built-in).

### 4. Autonomous
Executes automatically without human intervention.

### 5. Transparent
Code and state visible to everyone on the blockchain.

### 6. Trustless
No need to trust counterparty or intermediary.

### 7. Self-Verifying
Blockchain ensures execution according to code.

### 8. Self-Enforcing
Automatically enforces terms without external enforcement.

### 1. Access Control

```solidity
contract AccessControl {
    address public owner;
    mapping(address => bool) public admins;
    
    constructor() {
        owner = msg.sender;
    }
    
    modifier onlyOwner() {
        require(msg.sender == owner, "Not owner");
        _;
    }
    
    modifier onlyAdmin() {
        require(admins[msg.sender], "Not admin");
        _;
    }
    
    function addAdmin(address admin) public onlyOwner {
        admins[admin] = true;
    }
    
    function removeAdmin(address admin) public onlyOwner {
        admins[admin] = false;
    }
    
    function transferOwnership(address newOwner) public onlyOwner {
        require(newOwner != address(0), "Invalid address");
        owner = newOwner;
    }
}
```

### 2. Withdrawal Pattern

```solidity
contract WithdrawalPattern {
    mapping(address => uint) public pendingWithdrawals;
    
    function allowWithdrawal(address user, uint amount) internal {
        pendingWithdrawals[user] += amount;
    }
    
    function withdraw() public {
        uint amount = pendingWithdrawals[msg.sender];
        require(amount > 0, "No funds to withdraw");
        
        // Set to zero before transfer (prevent reentrancy)
        pendingWithdrawals[msg.sender] = 0;
        
        (bool success, ) = msg.sender.call{value: amount}("");
        require(success, "Transfer failed");
    }
}
```

## Real-World Smart Contract Examples

### 1. Simple Token

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract SimpleToken {
    string public name = "SimpleToken";
    string public symbol = "SIM";
    uint8 public decimals = 18;
    uint256 public totalSupply;
    
    mapping(address => uint256) public balanceOf;
    mapping(address => mapping(address => uint256)) public allowance;
    
    event Transfer(address indexed from, address indexed to, uint256 value);
    event Approval(address indexed owner, address indexed spender, uint256 value);
    
    constructor(uint256 initialSupply) {
        totalSupply = initialSupply * 10 ** uint256(decimals);
        balanceOf[msg.sender] = totalSupply;
    }
    
    function transfer(address to, uint256 value) public returns (bool) {
        require(balanceOf[msg.sender] >= value, "Insufficient balance");
        
        balanceOf[msg.sender] -= value;
        balanceOf[to] += value;
        
        emit Transfer(msg.sender, to, value);
        return true;
    }
    
    function approve(address spender, uint256 value) public returns (bool) {
        allowance[msg.sender][spender] = value;
        emit Approval(msg.sender, spender, value);
        return true;
    }
    
    function transferFrom(address from, address to, uint256 value) public returns (bool) {
        require(balanceOf[from] >= value, "Insufficient balance");
        require(allowance[from][msg.sender] >= value, "Allowance exceeded");
        
        balanceOf[from] -= value;
        balanceOf[to] += value;
        allowance[from][msg.sender] -= value;
        
        emit Transfer(from, to, value);
        return true;
    }
}
```

### 2. Crowdfunding

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract Crowdfunding {
    address public creator;
    uint public goal;
    uint public deadline;
    uint public totalFunded;
    bool public goalReached;
    bool public campaignClosed;
    
    mapping(address => uint) public contributions;
    
    event FundingReceived(address contributor, uint amount, uint totalFunded);
    event GoalReached(uint totalFunded);
    event FundWithdrawal(uint amount);
    
    constructor(uint _goal, uint durationInDays) {
        creator = msg.sender;
        goal = _goal;
        deadline = block.timestamp + (durationInDays * 1 days);
    }
    
    function contribute() public payable {
        require(block.timestamp < deadline, "Campaign ended");
        require(!campaignClosed, "Campaign closed");
        
        contributions[msg.sender] += msg.value;
        totalFunded += msg.value;
        
        emit FundingReceived(msg.sender, msg.value, totalFunded);
        
        if (totalFunded >= goal) {
            goalReached = true;
            emit GoalReached(totalFunded);
        }
    }
    
    function checkGoalReached() public {
        require(block.timestamp >= deadline, "Campaign still active");
        
        if (totalFunded >= goal) {
            goalReached = true;
            emit GoalReached(totalFunded);
        }
        campaignClosed = true;
    }
    
    function withdrawFunds() public {
        require(msg.sender == creator, "Only creator can withdraw");
        require(goalReached, "Goal not reached");
        require(campaignClosed, "Campaign not closed");
        
        uint amount = address(this).balance;
        payable(creator).transfer(amount);
        
        emit FundWithdrawal(amount);
    }
    
    function refund() public {
        require(block.timestamp >= deadline, "Campaign still active");
        require(!goalReached, "Goal was reached");
        require(contributions[msg.sender] > 0, "No contribution");
        
        uint amount = contributions[msg.sender];
        contributions[msg.sender] = 0;
        
        payable(msg.sender).transfer(amount);
    }
}
```

### 3. Multi-Signature Wallet

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract MultiSigWallet {
    event Deposit(address indexed sender, uint amount);
    event Submit(uint indexed txId);
    event Approve(address indexed owner, uint indexed txId);
    event Revoke(address indexed owner, uint indexed txId);
    event Execute(uint indexed txId);
    
    struct Transaction {
        address to;
        uint value;
        bytes data;
        bool executed;
    }
    
    address[] public owners;
    mapping(address => bool) public isOwner;
    uint public required;
    
    Transaction[] public transactions;
    mapping(uint => mapping(address => bool)) public approved;
    
    modifier onlyOwner() {
        require(isOwner[msg.sender], "Not owner");
        _;
    }
    
    modifier txExists(uint _txId) {
        require(_txId < transactions.length, "Transaction does not exist");
        _;
    }
    
    modifier notExecuted(uint _txId) {
        require(!transactions[_txId].executed, "Transaction already executed");
        _;
    }
    
    modifier notApproved(uint _txId) {
        require(!approved[_txId][msg.sender], "Transaction already approved");
        _;
    }
    
    constructor(address[] memory _owners, uint _required) {
        require(_owners.length > 0, "Owners required");
        require(_required > 0 && _required <= _owners.length, "Invalid required number");
        
        for (uint i = 0; i < _owners.length; i++) {
            address owner = _owners[i];
            require(owner != address(0), "Invalid owner");
            require(!isOwner[owner], "Owner not unique");
            
            isOwner[owner] = true;
            owners.push(owner);
        }
        
        required = _required;
    }
    
    receive() external payable {
        emit Deposit(msg.sender, msg.value);
    }
    
    function submit(address _to, uint _value, bytes calldata _data) external onlyOwner {
        transactions.push(Transaction({
            to: _to,
            value: _value,
            data: _data,
            executed: false
        }));
        
        emit Submit(transactions.length - 1);
    }
    
    function approve(uint _txId) 
        external 
        onlyOwner 
        txExists(_txId) 
        notExecuted(_txId) 
        notApproved(_txId) 
    {
        approved[_txId][msg.sender] = true;
        emit Approve(msg.sender, _txId);
    }
    
    function execute(uint _txId) external txExists(_txId) notExecuted(_txId) {
        require(_getApprovalCount(_txId) >= required, "Approvals < required");
        
        Transaction storage transaction = transactions[_txId];
        transaction.executed = true;
        
        (bool success, ) = transaction.to.call{value: transaction.value}(transaction.data);
        require(success, "Transaction failed");
        
        emit Execute(_txId);
    }
    
    function _getApprovalCount(uint _txId) private view returns (uint count) {
        for (uint i = 0; i < owners.length; i++) {
            if (approved[_txId][owners[i]]) {
                count += 1;
            }
        }
    }
}
```


## Smart Contract Security

### Common Vulnerabilities

#### 1. Reentrancy

Attacker recursively calls vulnerable function before state updates.

```solidity
// VULNERABLE
contract VulnerableBank {
    mapping(address => uint) public balances;
    
    function withdraw(uint amount) public {
        require(balances[msg.sender] >= amount);
        
        // DANGER: External call before state update
        (bool success, ) = msg.sender.call{value: amount}("");
        require(success);
        
        balances[msg.sender] -= amount;  // Too late!
    }
}

// SECURE: Checks-Effects-Interactions pattern
contract SecureBank {
    mapping(address => uint) public balances;
    
    function withdraw(uint amount) public {
        require(balances[msg.sender] >= amount);
        
        // Update state BEFORE external call
        balances[msg.sender] -= amount;
        
        (bool success, ) = msg.sender.call{value: amount}("");
        require(success);
    }
}

// SECURE: Using ReentrancyGuard
import "@openzeppelin/contracts/security/ReentrancyGuard.sol";

contract SecureBankWithGuard is ReentrancyGuard {
    mapping(address => uint) public balances;
    
    function withdraw(uint amount) public nonReentrant {
        require(balances[msg.sender] >= amount);
        
        balances[msg.sender] -= amount;
        
        (bool success, ) = msg.sender.call{value: amount}("");
        require(success);
    }
}
```

#### 2. Integer Overflow/Underflow

```solidity
// VULNERABLE (Solidity < 0.8.0)
contract VulnerableCounter {
    uint8 public count = 255;
    
    function increment() public {
        count += 1;  // Overflows to 0!
    }
}

// SECURE (Solidity >= 0.8.0 has built-in checks)
contract SecureCounter {
    uint8 public count = 255;
    
    function increment() public {
        count += 1;  // Reverts on overflow
    }
}

// SECURE (Using SafeMath for older Solidity)
import "@openzeppelin/contracts/utils/math/SafeMath.sol";

contract SecureCounterOld {
    using SafeMath for uint8;
    uint8 public count = 255;
    
    function increment() public {
        count = count.add(1);  // Reverts on overflow
    }
}
```

#### 4. Front-Running

```solidity
// VULNERABLE: Predictable outcome
contract VulnerableLottery {
    function play() public payable returns (bool) {
        // Miner can see this and front-run
        if (block.timestamp % 2 == 0) {
            payable(msg.sender).transfer(msg.value * 2);

// MITIGATION: Commit-Reveal scheme
contract SecureLottery {
    mapping(address => bytes32) public commits;
    
    // Step 1: Commit hash
    function commit(bytes32 hash) public {
        commits[msg.sender] = hash;
    }
    
    // Step 2: Reveal after commit is mined
    function reveal(uint number, bytes32 salt) public payable {
        require(keccak256(abi.encodePacked(number, salt)) == commits[msg.sender]);
        
        // Now use the number
        if (number % 2 == 0) {
            payable(msg.sender).transfer(msg.value * 2);
        }
    }
}
```

### Security Best Practices

```solidity
// COMPREHENSIVE SECURE CONTRACT
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "@openzeppelin/contracts/security/ReentrancyGuard.sol";
import "@openzeppelin/contracts/security/Pausable.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

contract SecureVault is ReentrancyGuard, Pausable, Ownable {
    mapping(address => uint256) private balances;
    
    event Deposit(address indexed user, uint256 amount);
    event Withdrawal(address indexed user, uint256 amount);
    
    // 1. Use ReentrancyGuard
    function deposit() external payable nonReentrant whenNotPaused {
        require(msg.value > 0, "Must deposit something");
        
        balances[msg.sender] += msg.value;
        emit Deposit(msg.sender, msg.value);
    }
    
    // 2. Checks-Effects-Interactions pattern
    function withdraw(uint256 amount) external nonReentrant whenNotPaused {
        // Checks
        require(amount > 0, "Must withdraw something");
        require(balances[msg.sender] >= amount, "Insufficient balance");
        
        // Effects (update state before external call)
        balances[msg.sender] -= amount;
        
        // Interactions (external call last)
        (bool success, ) = payable(msg.sender).call{value: amount}("");
        require(success, "Transfer failed");
        
        emit Withdrawal(msg.sender, amount);
    }
    
    // 3. Emergency pause
    function pause() external onlyOwner {
        _pause();
    }
    
    function unpause() external onlyOwner {
        _unpause();
    }
    
    // 4. View functions for transparency
    function getBalance(address user) external view returns (uint256) {
        return balances[user];
    }
    
    // 5. Explicit function visibility
    function getContractBalance() external view returns (uint256) {
        return address(this).balance;
    }
}
```

## Gas Optimization

### Tips for Reducing Gas Costs

```solidity
contract GasOptimization {
    // 1. Use appropriate data types
    uint8 public smallNumber;   // Cheaper than uint256 in structs/arrays
    uint256 public largeNumber; // Use uint256 for single storage slots
    
    // 2. Pack variables (order matters!)
    struct Efficient {
        uint128 a;  // 16 bytes
        uint128 b;  // 16 bytes - packed in same slot!
        address c;  // 20 bytes - new slot
    }
    
    struct Inefficient {
        uint256 a;  // 32 bytes - full slot
        uint128 b;  // 16 bytes - new slot (wastes 16 bytes)
        address c;  // 20 bytes - new slot
    }
    
    // 3. Use calldata instead of memory for read-only function parameters
    function processData(uint[] calldata data) external pure returns (uint) {
        // calldata is cheaper than memory
        return data[0];
    }
    
    // 4. Cache storage variables in memory
    uint[] public numbers;
    
    function sumBad() public view returns (uint) {
        uint sum = 0;
        for (uint i = 0; i < numbers.length; i++) {
            sum += numbers[i];  // SLOAD every iteration (expensive!)
        }
        return sum;
    }
    
    function sumGood() public view returns (uint) {
        uint[] memory nums = numbers;  // Cache in memory
        uint sum = 0;
        for (uint i = 0; i < nums.length; i++) {
            sum += nums[i];  // Memory access (cheap!)
        }
        return sum;
    }
    
    // 5. Use events instead of storage for data you don't need to query on-chain
    event DataStored(uint indexed id, string data);
    
    function storeData(uint id, string calldata data) external {
        emit DataStored(id, data);  // Much cheaper than storage!
    }
    
    // 6. Use unchecked for operations that won't overflow
    function increment(uint x) public pure returns (uint) {
        unchecked {
            return x + 1;  // Skip overflow check (use carefully!)
        }
    }
    
    // 7. Short-circuit with require
    function checkConditions(bool condition1, bool condition2) public pure {
        require(condition1 && condition2, "Conditions not met");
        // If condition1 is false, condition2 is never evaluated
    }
}
```

## Upgradeable Contracts

### Proxy Pattern

```solidity
// Implementation contract
contract ImplementationV1 {
    uint public value;
    
    function setValue(uint _value) public {
        value = _value;
    }
}

// Proxy contract
contract Proxy {
    address public implementation;
    address public admin;
    
    constructor(address _implementation) {
        implementation = _implementation;
        admin = msg.sender;
    }
    
    function upgrade(address newImplementation) external {
        require(msg.sender == admin, "Not admin");
        implementation = newImplementation;
    }
    
    fallback() external payable {
        address impl = implementation;
        
        assembly {
            // Copy msg.data
            calldatacopy(0, 0, calldatasize())
            
            // Delegate call to implementation
            let result := delegatecall(gas(), impl, 0, calldatasize(), 0, 0)
            
            // Copy return data
            returndatacopy(0, 0, returndatasize())
            
            // Return or revert
            switch result
            case 0 { revert(0, returndatasize()) }
            default { return(0, returndatasize()) }
        }
    }
}
```

## Interacting with Smart Contracts

### Using Web3.js

```javascript
const Web3 = require('web3');
const web3 = new Web3('https://mainnet.infura.io/v3/YOUR_INFURA_KEY');

// Contract ABI and address
const abi = [...];  // Contract ABI
const address = '0x...';  // Contract address

// Create contract instance
const contract = new web3.eth.Contract(abi, address);

// Read data (call)
const balance = await contract.methods.balanceOf('0xUserAddress').call();
console.log('Balance:', balance);

// Write data (send transaction)
const accounts = await web3.eth.getAccounts();
await contract.methods.transfer('0xRecipient', '1000000000000000000')
  .send({ from: accounts[0] });

// Listen to events
contract.events.Transfer({
  filter: { from: '0xUserAddress' },
  fromBlock: 'latest'
})
.on('data', (event) => {
  console.log('Transfer event:', event.returnValues);
})
.on('error', console.error);
```

### Using Ethers.js

```javascript
const { ethers } = require('ethers');

// Connect to provider
const provider = new ethers.providers.JsonRpcProvider('https://mainnet.infura.io/v3/YOUR_INFURA_KEY');

// Connect wallet
const wallet = new ethers.Wallet('PRIVATE_KEY', provider);

// Contract instance
const abi = [...];
const address = '0x...';
const contract = new ethers.Contract(address, abi, wallet);

// Read data
const balance = await contract.balanceOf('0xUserAddress');
console.log('Balance:', ethers.utils.formatEther(balance));

// Write data
const tx = await contract.transfer('0xRecipient', ethers.utils.parseEther('1.0'));

// Listen to events

  console.log(`Transfer: ${from} -> ${to}: ${ethers.utils.formatEther(amount)} tokens`);
});
```
