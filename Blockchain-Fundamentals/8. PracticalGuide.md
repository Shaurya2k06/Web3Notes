### Write Smart Contract

Create `contracts/SimpleStorage.sol`:

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract SimpleStorage {
    uint256 private storedData;
    
    event DataStored(uint256 data, address indexed by);
    
    function set(uint256 x) public {
        storedData = x;
        emit DataStored(x, msg.sender);
    }
    
    function get() public view returns (uint256) {
        return storedData;
    }
}
```

### Write Tests

Create `test/SimpleStorage.test.js`:

```javascript
const { expect } = require("chai");
const { ethers } = require("hardhat");

describe("SimpleStorage", function () {
  let simpleStorage;
  let owner;
  let addr1;

  beforeEach(async function () {
    [owner, addr1] = await ethers.getSigners();
    
    const SimpleStorage = await ethers.getContractFactory("SimpleStorage");
    simpleStorage = await SimpleStorage.deploy();
    await simpleStorage.deployed();
  });

  describe("Deployment", function () {
    it("Should start with 0", async function () {
      expect(await simpleStorage.get()).to.equal(0);
    });
  });

  describe("Storage", function () {
    it("Should store the value", async function () {
      await simpleStorage.set(42);
      expect(await simpleStorage.get()).to.equal(42);
    });

    it("Should emit DataStored event", async function () {
      await expect(simpleStorage.set(42))
        .to.emit(simpleStorage, "DataStored")
        .withArgs(42, owner.address);
    });

    it("Should allow different users to set values", async function () {
      await simpleStorage.connect(addr1).set(100);
      expect(await simpleStorage.get()).to.equal(100);
    });
  });
});
```

### Deploy to Local Network

Create `scripts/deploy.js`:

```javascript
const hre = require("hardhat");

async function main() {
  const SimpleStorage = await hre.ethers.getContractFactory("SimpleStorage");
  const simpleStorage = await SimpleStorage.deploy();

  await simpleStorage.deployed();

  console.log("SimpleStorage deployed to:", simpleStorage.address);
}

main()
  .then(() => process.exit(0))
  .catch((error) => {
    console.error(error);
    process.exit(1);
  });
```

```bash
# Start local node
npx hardhat node

# Deploy (in another terminal)
npx hardhat run scripts/deploy.js --network localhost
```

## Building a DApp Frontend

### Setup React App

```bash
npx create-react-app my-dapp
cd my-dapp
npm install ethers
```

### Create Contract Interaction Component

`src/App.js`:

```javascript
import { useState, useEffect } from 'react';
import { ethers } from 'ethers';
import './App.css';

// Replace with your contract address and ABI
const CONTRACT_ADDRESS = "0x...";
const CONTRACT_ABI = [
  "function get() view returns (uint256)",
  "function set(uint256 x)",
  "event DataStored(uint256 data, address indexed by)"
];

function App() {
  const [currentValue, setCurrentValue] = useState(0);
  const [newValue, setNewValue] = useState('');
  const [account, setAccount] = useState('');
  const [contract, setContract] = useState(null);

  useEffect(() => {
    connectWallet();
  }, []);

  async function connectWallet() {
    if (typeof window.ethereum !== 'undefined') {
      try {
        // Request account access
        await window.ethereum.request({ method: 'eth_requestAccounts' });
        
        const provider = new ethers.providers.Web3Provider(window.ethereum);
        const signer = provider.getSigner();
        const address = await signer.getAddress();
        
        setAccount(address);
        
        // Create contract instance
        const contractInstance = new ethers.Contract(
          CONTRACT_ADDRESS,
          CONTRACT_ABI,
          signer
        );
        setContract(contractInstance);
        
        // Get current value
        const value = await contractInstance.get();
        setCurrentValue(value.toString());
        
        // Listen for events
        contractInstance.on("DataStored", (data, by) => {
          console.log(`Data stored: ${data} by ${by}`);
          setCurrentValue(data.toString());
        });
        
      } catch (error) {
        console.error("Error connecting wallet:", error);
      }
    } else {
      alert('Please install MetaMask!');
    }
  }

  async function storeValue() {
    if (!contract) return;
    
    try {
      const tx = await contract.set(newValue);
      console.log("Transaction sent:", tx.hash);
      
      await tx.wait();
      console.log("Transaction confirmed!");
      
      setNewValue('');
    } catch (error) {
      console.error("Error storing value:", error);
    }
  }

  return (
    <div className="App">
      <header className="App-header">
        <h1>Simple Storage DApp</h1>
        
        {account ? (
          <p>Connected: {account.slice(0, 6)}...{account.slice(-4)}</p>
        ) : (
          <button onClick={connectWallet}>Connect Wallet</button>
        )}
        
        <div className="storage-section">
          <h2>Current Value: {currentValue}</h2>
          
          <div className="input-section">
            <input
              type="number"
              value={newValue}
              onChange={(e) => setNewValue(e.target.value)}
              placeholder="Enter new value"
            />
            <button onClick={storeValue}>Store Value</button>
          </div>
        </div>
      </header>
    </div>
  );
}

export default App;
```

### Run DApp

```bash
npm start
```

## Common Development Patterns

### 1. Reading Blockchain Data

```javascript
// Get block number
const blockNumber = await provider.getBlockNumber();

// Get block details
const block = await provider.getBlock(blockNumber);

// Get transaction
const tx = await provider.getTransaction(txHash);

// Get transaction receipt
const receipt = await provider.getTransactionReceipt(txHash);

// Get account balance
const balance = await provider.getBalance(address);
console.log(ethers.utils.formatEther(balance));

// Get gas price
const gasPrice = await provider.getGasPrice();
```

### 2. Sending Transactions

```javascript
// Send ETH
const tx = await signer.sendTransaction({
  to: recipientAddress,
  value: ethers.utils.parseEther("1.0")
});
await tx.wait();

// Call contract function
const tx = await contract.transfer(recipientAddress, amount);
await tx.wait();

// Call with custom gas
const tx = await contract.transfer(recipientAddress, amount, {
  gasLimit: 100000,
  gasPrice: ethers.utils.parseUnits("50", "gwei")
});
```

### 3. Listening to Events

```javascript
// Listen to specific event
contract.on("Transfer", (from, to, amount) => {
  console.log(`Transfer: ${from} -> ${to}: ${amount}`);
});

// Listen to all events
contract.on("*", (event) => {
  console.log("Event:", event);
});

// Query past events
const filter = contract.filters.Transfer(null, myAddress);
const events = await contract.queryFilter(filter, fromBlock, toBlock);
```

### 4. Error Handling

```javascript
try {
  const tx = await contract.someFunction();
  await tx.wait();
} catch (error) {
  if (error.code === 'INSUFFICIENT_FUNDS') {
    console.error("Insufficient funds");
  } else if (error.code === 'UNPREDICTABLE_GAS_LIMIT') {
    console.error("Transaction will likely fail");
  } else if (error.message.includes("user rejected")) {
    console.error("User rejected transaction");
  } else {
    console.error("Error:", error.message);
  }
}
```

## Security Best Practices

### Smart Contract Security

```solidity
// 1. Use latest Solidity version
pragma solidity ^0.8.19;

// 2. Import OpenZeppelin contracts
import "@openzeppelin/contracts/security/ReentrancyGuard.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

// 3. Use SafeMath (built-in for 0.8+)
// No need for SafeMath library in Solidity 0.8+

// 4. Checks-Effects-Interactions pattern
function withdraw(uint amount) public {
    // Checks
    require(balances[msg.sender] >= amount);
    
    // Effects
    balances[msg.sender] -= amount;
    
    // Interactions
    (bool success, ) = msg.sender.call{value: amount}("");
    require(success);
}

// 5. Use modifiers for access control
modifier onlyOwner() {
    require(msg.sender == owner);
    _;
}

// 6. Emit events for important state changes
event Withdrawal(address indexed user, uint amount);

// 7. Use require for validation
require(amount > 0, "Amount must be positive");

// 8. Avoid delegatecall to untrusted contracts
// 9. Be careful with external calls
// 10. Test thoroughly
```

### 1. Use Console.log in Solidity

```solidity
import "hardhat/console.sol";

function debug() public {
    console.log("Value:", someValue);
    console.log("Sender:", msg.sender);
}
```

### 2. Hardhat Network Forking

```javascript
// hardhat.config.js
networks: {
  hardhat: {
    forking: {
      url: "https://eth-mainnet.alchemyapi.io/v2/YOUR_KEY",
      blockNumber: 14390000

  }

```

### 3. Transaction Simulation

```javascript
// Estimate gas before sending
const gasEstimate = await contract.estimateGas.someFunction(params);
console.log("Estimated gas:", gasEstimate.toString());

// Call (simulate) before sending transaction
const result = await contract.callStatic.someFunction(params);
console.log("Simulated result:", result);
```

### 4. Event Debugging

```javascript
// Get transaction receipt with events
const tx = await contract.someFunction();
const receipt = await tx.wait();

// Parse events
receipt.events.forEach((event) => {
  console.log("Event:", event.event);
  console.log("Args:", event.args);
});
```
